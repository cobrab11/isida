# ---------------------------------------------------------------
#                      Регекспы для цензора
# ---------------------------------------------------------------

# вариации на слово хуй
.*[ ]ху{1,}й[ ].*
.*[^пси]ху{1,}й.*
ху{1,}й.*

# вариации на слово пизда
.*[ ]п[ие]зд[ае].*
п[ие]зд[ае].*

# вариации на слово бля\ебля
.*[^гре|^ру]бля{1,}.*
.*[ ][её]б[ ].*
[её]б.*
бля{1,}.*

# вариации на слово пидар
.*[^ски]п[ие]д[оа]р.*
п[ие]д[оа]р.*
